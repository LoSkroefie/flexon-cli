1. Basic Usage

C# Basic Usage


using FlexonCLI;
using System.Text.Json;

// Create a simple object
var data = new { Name = "John", Age = 30 };

// Serialize to FLEXON
var flexonData = Flexon.Serialize(data);
File.WriteAllBytes("data.flexon", flexonData);

// Deserialize from FLEXON
var readData = File.ReadAllBytes("data.flexon");
var result = Flexon.Deserialize(readData);

Console.WriteLine($"Name: {result.Name}, Age: {result.Age}");

Java Basic Usage


import com.flexon.FlexonCLI;
import com.flexon.FlexonObject;

public class BasicUsage {
    public static void main(String[] args) {
        // Create data object
        FlexonObject data = new FlexonObject();
        data.put("name", "John");
        data.put("age", 30);

        // Serialize to FLEXON
        byte[] flexonData = FlexonCLI.serialize(data);
        Files.write(Paths.get("data.flexon"), flexonData);

        // Deserialize from FLEXON
        byte[] readData = Files.readAllBytes(Paths.get("data.flexon"));
        FlexonObject result = FlexonCLI.deserialize(readData);

        System.out.println("Name: " + result.getString("name") + 
                         ", Age: " + result.getInt("age"));
    }
}

Python Basic Usage


from flexon import FlexonCLI

// Create data dictionary
data = {
    "name": "John",
    "age": 30
}

// Serialize to FLEXON
flexon_data = FlexonCLI.serialize(data)
with open("data.flexon", "wb") as f:
    f.write(flexon_data)

// Deserialize from FLEXON
with open("data.flexon", "rb") as f:
    read_data = f.read()
result = FlexonCLI.deserialize(read_data)

print(f"Name: {result['name']}, Age: {result['age']}")

JavaScript Basic Usage


const FlexonCLI = require('flexon-cli');
const fs = require('fs').promises;

async function basicUsage() {
    // Create data object
    const data = {
        name: "John",
        age: 30
    };

    // Serialize to FLEXON
    const flexonData = FlexonCLI.serialize(data);
    await fs.writeFile('data.flexon', flexonData);

    // Deserialize from FLEXON
    const readData = await fs.readFile('data.flexon');
    const result = FlexonCLI.deserialize(readData);

    console.log(`Name: ${result.name}, Age: ${result.age}`);
}

basicUsage().catch(console.error);

2. AI Features

Python AI Data Processing


from flexon import FlexonCLI, AIFeatures

// Sample data with text for analysis
data = {
    "text": "The quick brown fox jumps over the lazy dog",
    "metadata": {
        "language": "en",
        "priority": "high"
    }
}

// Initialize AI features
ai = AIFeatures()

// Analyze sentiment and extract keywords
analysis = ai.analyze_text(data["text"])

// Add AI analysis to data
data["analysis"] = {
    "sentiment": analysis.sentiment,
    "keywords": analysis.keywords,
    "summary": analysis.summary
}

// Serialize enriched data
flexon_data = FlexonCLI.serialize(data)
with open("ai_processed.flexon", "wb") as f:
    f.write(flexon_data)

C# AI Integration


using FlexonCLI;
using FlexonCLI.AI;

// Initialize AI processor
var aiProcessor = new AIFeatures();

// Sample document data
var document = new {
    Content = "Important business document about AI integration",
    Metadata = new {
        Author = "Jane Smith",
        Date = DateTime.Now
    }
};

// Process with AI
var aiResult = aiProcessor.ProcessDocument(document);

// Combine original data with AI insights
var enrichedData = new {
    Original = document,
    AIInsights = new {
        Category = aiResult.Category,
        Keywords = aiResult.Keywords,
        Summary = aiResult.Summary,
        Importance = aiResult.ImportanceScore
    }
};

// Serialize enriched data
var flexonData = Flexon.Serialize(enrichedData);
File.WriteAllBytes("ai_enriched.flexon", flexonData);

3. Game State Management

Java Game State


import com.flexon.FlexonCLI;
import com.flexon.GameState;

public class GameStateExample {
    public static void main(String[] args) {
        // Create game state
        GameState state = new GameState();
        state.setPlayerPosition(new Vector3(10.5f, 0, 15.2f));
        state.setHealth(100);
        state.setInventory(new String[]{"Sword", "Shield", "Potion"});

        // Save game state
        byte[] stateData = FlexonCLI.serialize(state);
        Files.write(Paths.get("save.flexon"), stateData);

        // Load game state
        byte[] loadedData = Files.readAllBytes(Paths.get("save.flexon"));
        GameState loadedState = FlexonCLI.deserialize(loadedData, GameState.class);

        // Verify state
        System.out.println("Player position: " + loadedState.getPlayerPosition());
        System.out.println("Health: " + loadedState.getHealth());
        System.out.println("Inventory: " + 
            String.join(", ", loadedState.getInventory()));
    }
}

C# Game State


using FlexonCLI;
using UnityEngine;

public class GameStateManager
{
    public void SaveGameState(GameState state)
    {
        var stateData = Flexon.Serialize(state);
        File.WriteAllBytes("save.flexon", stateData);
    }

    public GameState LoadGameState()
    {
        var stateData = File.ReadAllBytes("save.flexon");
        return Flexon.Deserialize(stateData);
    }
}

public class GameState
{
    public Vector3 PlayerPosition { get; set; }
    public int Health { get; set; }
    public string[] Inventory { get; set; }
    public Dictionary Stats { get; set; }
}
